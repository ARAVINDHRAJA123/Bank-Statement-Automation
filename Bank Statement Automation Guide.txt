Part B: The Main Cleaning Engine
This block reads every page, merges broken text, and categorizes your spending.

Python
import pdfplumber
import pandas as pd

# 🚀 CONFIGURATION
FILENAME = "ACC statement sept 30 to feb 6_unlocked.pdf" # <-- Change to your filename
OUTPUT = "Organized_Expenses.xlsx"

def generate_report(pdf_name):
    all_data = []
    print("⏳ Processing 72 pages... please wait.")
    
    with pdfplumber.open(pdf_name) as pdf:
        for page in pdf.pages:
            table = page.extract_table()
            if table:
                all_data.extend(table[1:]) 

    # Creating the Table
    df = pd.DataFrame(all_data, columns=["Date", "Narration", "Ref", "Value_Dt", "Sent", "Received", "Balance"])

    # 🛠️ DATA CLEANING
    df['Date'] = df['Date'].replace('', None).ffill() # Fixes broken rows
    df['Sent'] = pd.to_numeric(df['Sent'].str.replace(',', ''), errors='coerce').fillna(0)
    df['Received'] = pd.to_numeric(df['Received'].str.replace(',', ''), errors='coerce').fillna(0)

    # 🏷️ CATEGORIZATION (Your Specific Brackets)
    def get_bracket(amt):
        if amt == 0: return "Received (Money In)"
        if amt < 100: return "Under 100 rs"
        if amt < 500: return "Under 500 rs"
        if amt < 1000: return "Under 1000 rs"
        if amt < 5000: return "Under 5000 rs"
        return "Above 5000 rs"

    df['Bracket'] = df['Sent'].apply(get_bracket)
    
    # 🏁 FINAL SORT & EXPORT
    # Sorts so the biggest "Above 5000" expenses appear first
    df = df.sort_values(by=["Sent"], ascending=False)
    df.to_excel(OUTPUT, index=False)
    
    print(f"✨ SUCCESS! '{OUTPUT}' has been created.")

generate_report(FILENAME)
